## 🔄 API 1: POST /photos/batch-upload - 사진 업로드 및 분석

> 여러 사진을 한 번에 업로드하고, EXIF 파싱 후 시간대별로 다이어리에 분류한 뒤, **LLM 분석까지 동기로 수행**합니다.
>
> **아키텍처: 동기 + 내부 병렬 처리 (1안)**
>
> - 업로드 API에서 분석까지 전부 수행
> - `asyncio.gather()`로 LLM 호출 병렬 처리
> - 응답 시간: 5-10초 (사진 개수와 무관)
> - 프론트 로직 단순 (폴링 불필요)

```python
from PIL import Image
from PIL.ExifTags import TAGS, GPSTAGS
from datetime import datetime
from typing import TypedDict
from io import BytesIO

class ExifData(TypedDict):
    taken_at: datetime | None
    latitude: float | None
    longitude: float | None

def extract_exif_data(file) -> ExifData:
    """
    이미지 파일에서 EXIF 데이터를 추출합니다.

    Returns:
        {
            "taken_at": datetime 또는 None,
            "latitude": float 또는 None,
            "longitude": float 또는 None
        }
    """
    result: ExifData = {
        "taken_at": None,
        "latitude": None,
        "longitude": None
    }

    try:
        image = Image.open(file)
        exif_data = image._getexif()

        if exif_data is None:
            return result

        # EXIF 태그 파싱
        for tag_id, value in exif_data.items():
            tag = TAGS.get(tag_id, tag_id)

            # 촬영 시간
            if tag == "DateTimeOriginal":
                try:
                    result["taken_at"] = datetime.strptime(value, "%Y:%m:%d %H:%M:%S")
                except ValueError:
                    pass

            # GPS 정보
            if tag == "GPSInfo":
                gps_data = {}
                for gps_tag_id, gps_value in value.items():
                    gps_tag = GPSTAGS.get(gps_tag_id, gps_tag_id)
                    gps_data[gps_tag] = gps_value

                # 위도
                if "GPSLatitude" in gps_data and "GPSLatitudeRef" in gps_data:
                    lat = _convert_to_degrees(gps_data["GPSLatitude"])
                    if gps_data["GPSLatitudeRef"] == "S":
                        lat = -lat
                    result["latitude"] = lat

                # 경도
                if "GPSLongitude" in gps_data and "GPSLongitudeRef" in gps_data:
                    lng = _convert_to_degrees(gps_data["GPSLongitude"])
                    if gps_data["GPSLongitudeRef"] == "W":
                        lng = -lng
                    result["longitude"] = lng

    except Exception:
        pass  # EXIF 파싱 실패 시 기본값 반환

    return result

def _convert_to_degrees(value) -> float:
    """GPS 좌표를 도(degree) 단위로 변환"""
    d, m, s = value
    return float(d) + float(m) / 60 + float(s) / 3600
```

from datetime import datetime

def classify_time_type(taken_at: datetime | None) -> str:
    """
    촬영 시간을 기준으로 끼니 타입을 분류합니다.

    - 05:00 ~ 10:00: breakfast
    - 10:00 ~ 14:00: lunch
    - 14:00 ~ 17:00: snack
    - 17:00 ~ 22:00: dinner
    - 그 외: snack

    taken_at이 None이면 기본값 'lunch' 반환
    """
    if taken_at is None:
        return "lunch"

    hour = taken_at.hour
    if 5 <= hour < 10:
        return "breakfast"
    elif 10 <= hour < 14:
        return "lunch"
    elif 14 <= hour < 17:
        return "snack"
    elif 17 <= hour < 22:
        return "dinner"
    else:
        return "snack"
```

import uuid
from pathlib import Path
from fastapi import UploadFile

UPLOAD_DIR = Path("data/photos")

async def save_uploaded_file(file: UploadFile) -> str:
    """
    업로드된 파일을 저장하고 URL을 반환합니다.

    Returns:
        저장된 파일의 URL (예: "/data/photos/{uuid}.jpg")
    """
    UPLOAD_DIR.mkdir(parents=True, exist_ok=True)

    ext = Path(file.filename or "image.jpg").suffix or ".jpg"
    filename = f"{uuid.uuid4()}{ext}"
    filepath = UPLOAD_DIR / filename

    content = await file.read()
    filepath.write_bytes(content)
    await file.seek(0)  # 파일 포인터 리셋

    return f"/data/photos/{filename}"
```

from app.core.config import settings

async def search_nearby_restaurants(
    latitude: float,
    longitude: float,
    radius: int = 500
) -> list[dict]:
    """GPS 좌표 주변의 음식점을 검색합니다."""

    headers = {
        "Authorization": f"KakaoAK {settings.KAKAO_REST_API_KEY}"
    }
    params = {
        "query": "음식점",
        "x": str(longitude),
        "y": str(latitude),
        "radius": radius,
        "category_group_code": "FD6"
    }

    async with httpx.AsyncClient() as client:
        response = await client.get(url, headers=headers, params=params)
        response.raise_for_status()
        data = response.json()

        return [
            {
                "name": place["place_name"],
                "address": place["address_name"],
                "category": place["category_name"],
                "distance": int(place["distance"])
            }
            for place in data.get("documents", [])
        ]

import json
import google.generativeai as genai
from app.core.config import settings

genai.configure(api_key=settings.GEMINI_API_KEY)

async def analyze_food_image(image_path: str) -> dict:
    """이미지를 분석하여 음식 정보를 추출합니다."""
    model = genai.GenerativeModel("gemini-1.5-flash")

    prompt = """
    이 음식 사진을 분석해주세요.
    다음 정보를 JSON 형식으로만 반환해주세요 (다른 텍스트 없이):
    {
        "food_category": "음식 카테고리 (한식, 중식, 일식, 양식, 카페 등)",
        "menus": ["추정되는 메뉴 이름 리스트"],
        "keywords": ["음식을 설명하는 키워드 3~5개"]
    }
    """

    with open(image_path, "rb") as f:
        image_data = f.read()

    response = await model.generate_content_async([prompt, image_data])

    # JSON 파싱 (마크다운 코드블록 제거)
    text = response.text.strip()
    if text.startswith("```"):
        text = text.split("```")[1]
        if text.startswith("json"):
            text = text[4:]

    return json.loads(text)
````

### 1.5 분석 서비스

**파일:** `app/services/analysis_service.py`

```python
import asyncio
from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select

from app.models import Photo, PhotoAnalysisResult, DiaryAnalysis
from app.services.llm_service import analyze_food_image
from app.services.kakao_map_service import search_nearby_restaurants

async def analyze_photo(db: AsyncSession, photo_id: int) -> PhotoAnalysisResult | None:
    """
    사진 하나를 분석합니다.
    타임아웃 시 None 반환 (분석 실패해도 업로드는 성공)
    """
    try:
        return await asyncio.wait_for(
            _analyze_photo_internal(db, photo_id),
            timeout=30  # 30초 타임아웃
        )
    except asyncio.TimeoutError:
        return None
    except Exception:
        return None  # 분석 실패 시 None 반환

async def _analyze_photo_internal(db: AsyncSession, photo_id: int) -> PhotoAnalysisResult:
    """사진 분석 내부 로직"""
    photo = await db.get(Photo, photo_id)
    if not photo:
        raise ValueError("Photo not found")

    # 1. LLM 분석
    # image_url이 로컬 경로인 경우 처리
    image_path = photo.image_url
    if image_path.startswith("/"):
        image_path = image_path.lstrip("/")

    llm_result = await analyze_food_image(image_path)

    # 2. GPS가 있으면 Kakao Map 검색
    restaurant_candidates = []
    if photo.taken_location:
        lat, lng = map(float, photo.taken_location.split(","))
        nearby = await search_nearby_restaurants(lat, lng)
        restaurant_candidates = [
            {"name": r["name"], "confidence": 0.8, "address": r["address"]}
            for r in nearby[:5]
        ]

    # 3. PhotoAnalysisResult 저장
    result = PhotoAnalysisResult(
        photo_id=photo_id,
        food_category=llm_result.get("food_category", "기타"),
        restaurant_name_candidates=restaurant_candidates,
        menu_candidates=[{"name": m} for m in llm_result.get("menus", [])],
        keywords=llm_result.get("keywords", []),
        raw_response=str(llm_result)
    )
    db.add(result)
    await db.commit()
    await db.refresh(result)

    return result

async def aggregate_photo_analysis_to_diary(db: AsyncSession, diary_id: int) -> None:
    """
    다이어리에 속한 모든 사진의 분석 결과를 집계하여
    DiaryAnalysis 테이블에 저장합니다.
    """
    # 다이어리의 모든 사진 분석 결과 조회
    stmt = (
        select(PhotoAnalysisResult)
        .join(Photo)
        .where(Photo.diary_id == diary_id)
    )
    result = await db.execute(stmt)
    photo_results = list(result.scalars().all())

    if not photo_results:
        return

    # 집계: 식당 후보
    restaurant_candidates = []
    seen_restaurants = set()
    for pr in photo_results:
        for rc in (pr.restaurant_name_candidates or []):
            if rc["name"] not in seen_restaurants:
                seen_restaurants.add(rc["name"])
                restaurant_candidates.append(rc)

    # 집계: 카테고리 후보
    category_candidates = list(set(
        pr.food_category for pr in photo_results if pr.food_category
    ))

    # 집계: 메뉴 후보
    menu_candidates = []
    seen_menus = set()
    for pr in photo_results:
        for mc in (pr.menu_candidates or []):
            if mc["name"] not in seen_menus:
                seen_menus.add(mc["name"])
                menu_candidates.append(mc["name"])

    # DiaryAnalysis upsert
    existing = await db.get(DiaryAnalysis, diary_id)
    if existing:
        existing.restaurant_candidates = restaurant_candidates
        existing.category_candidates = category_candidates
        existing.menu_candidates = menu_candidates
    else:
        diary_analysis = DiaryAnalysis(
            diary_id=diary_id,
            restaurant_candidates=restaurant_candidates,
            category_candidates=category_candidates,
            menu_candidates=menu_candidates
        )
        db.add(diary_analysis)

    await db.commit()
```

from sqlalchemy.ext.asyncio import AsyncSession
from sqlalchemy import select
from datetime import date
from uuid import UUID
from app.models import Diary

async def get_or_create_diary(
    db: AsyncSession,
    user_id: UUID,
    diary_date: date,
    time_type: str
) -> Diary:
    """
    다이어리를 조회하거나 생성합니다 (upsert).
    user_id + date + time_type이 유니크 제약이므로 중복 방지.
    """
    stmt = select(Diary).where(
        Diary.user_id == user_id,
        Diary.diary_date == diary_date,
        Diary.time_type == time_type
    )
    result = await db.execute(stmt)
    diary = result.scalar_one_or_none()

    if diary is None:
        diary = Diary(
            user_id=user_id,
            diary_date=diary_date,
            time_type=time_type,
            photo_count=0
        )
        db.add(diary)
        await db.commit()
        await db.refresh(diary)

    return diary
```

**파일:** `app/services/photo_service.py`

```python
import asyncio
from fastapi import UploadFile
from sqlalchemy.ext.asyncio import AsyncSession
from datetime import date
from uuid import UUID

from app.models import Photo
from app.schemas.photo import PhotoUploadResult
from app.utils.exif_parser import extract_exif_data
from app.utils.time_classifier import classify_time_type
from app.utils.file_storage import save_uploaded_file
from app.services.diary_service import get_or_create_diary
from app.services.analysis_service import analyze_photo, aggregate_photo_analysis_to_diary

async def batch_upload_photos(
    db: AsyncSession,
    user_id: UUID,
    target_date: date,
    files: list[UploadFile]
) -> list[PhotoUploadResult]:
    """
    여러 사진을 한 번에 업로드하고 분석합니다.

    처리 순서:
    1. 파일 저장 + DB 저장 (순차)
    2. LLM 분석 (병렬 - asyncio.gather)
    3. DiaryAnalysis 집계
    """
    photo_infos = []  # [(photo, diary, time_type), ...]

    # ========================================
    # 1단계: 파일 저장 + DB 저장 (순차 처리)
    # ========================================
    for file in files:
        # 1. EXIF 파싱
        exif_data = extract_exif_data(file.file)
        await file.seek(0)

        # 2. 시간대 분류
        time_type = classify_time_type(exif_data["taken_at"])

        # 3. Diary upsert
        diary = await get_or_create_diary(db, user_id, target_date, time_type)

        # 4. 파일 저장
        image_url = await save_uploaded_file(file)

        # 5. Photo 생성
        taken_location = None
        if exif_data["latitude"] and exif_data["longitude"]:
            taken_location = f"{exif_data['latitude']},{exif_data['longitude']}"

        photo = Photo(
            diary_id=diary.id,
            image_url=image_url,
            taken_at=exif_data["taken_at"],
            taken_location=taken_location,
        )
        db.add(photo)
        diary.photo_count = (diary.photo_count or 0) + 1

        await db.commit()
        await db.refresh(photo)

        photo_infos.append((photo, diary, time_type))

    # ========================================
    # 2단계: LLM 분석 (병렬 처리)
    # ========================================
    await asyncio.gather(*[
        analyze_photo(db, photo.id)
        for photo, _, _ in photo_infos
    ])

    # ========================================
    # 3단계: DiaryAnalysis 집계
    # ========================================
    # 같은 다이어리에 대해 중복 집계 방지
    diary_ids = set(diary.id for _, diary, _ in photo_infos)
    for diary_id in diary_ids:
        await aggregate_photo_analysis_to_diary(db, diary_id)

    # ========================================
    # 결과 반환
    # ========================================
    return [
        PhotoUploadResult(
            photo_id=photo.id,
            diary_id=diary.id,
            time_type=time_type,
        )
        for photo, diary, time_type in photo_infos
    ]
```

```python
from fastapi import APIRouter, Depends, File, UploadFile, Form, HTTPException, status
from sqlalchemy.ext.asyncio import AsyncSession
from typing import Annotated
from datetime import datetime

from app.core.database import get_session
from app.core.dependencies import get_current_user
from app.models import User
from app.schemas.photo import BatchUploadResponse
from app.services.photo_service import batch_upload_photos as batch_upload_photos_service

router = APIRouter(prefix="/photos", tags=["photos"])

@router.post("/batch-upload", response_model=BatchUploadResponse)
async def batch_upload_photos(
    date: Annotated[str, Form()],
    photos: Annotated[list[UploadFile], File()],
    db: AsyncSession = Depends(get_session),
    current_user: User = Depends(get_current_user)
):
    """
    여러 사진을 한 번에 업로드하고 분석합니다.

    - 업로드 + EXIF 파싱 + 시간대 분류 + 다이어리 생성
    - LLM 분석 (병렬 처리)
    - 응답 시간: 약 5-10초
    """
    # date 문자열을 date 객체로 변환
    try:
        target_date = datetime.strptime(date, "%Y-%m-%d").date()
    except ValueError:
        raise HTTPException(
            status_code=status.HTTP_400_BAD_REQUEST,
            detail="Invalid date format. Use YYYY-MM-DD"
        )

    # 파일 형식 검증
    for photo in photos:
        if not photo.content_type or not photo.content_type.startswith("image/"):
            raise HTTPException(
                status_code=status.HTTP_400_BAD_REQUEST,
                detail=f"Invalid file type: {photo.content_type}"
            )

    # 배치 업로드 + 분석 처리
    results = await batch_upload_photos_service(
        db, current_user.id, target_date, photos
    )

    return BatchUploadResponse(created=results)
```

## 🔄 API 2: GET /diaries/{date} - 다이어리 조회

> 특정 날짜의 모든 다이어리(끼니별)를 조회합니다.

### 2.1 API 스펙

```
GET /diaries/{date}
Authorization: Bearer <token>

Response:

    "date": "2026-01-19",
    "diaries": [
        {
            "diary_id": 12,
            "time_type": "lunch",
            "analysis_status": "processing",
            "restaurant_name": null,
            "category": null,
            "cover_photo_url": "...",
            "photos": [

### 2.2 서비스 레이어

**파일:** `app/services/diary_service.py` (추가)

```python
from sqlalchemy.orm import selectinload

async def get_diaries_by_date(
    db: AsyncSession,
    user_id: UUID,
    target_date: date
) -> list[Diary]:
    """특정 날짜의 모든 다이어리 조회"""
    stmt = (
        select(Diary)
        .options(
            selectinload(Diary.photos),
            selectinload(Diary.analysis),
            selectinload(Diary.cover_photo)
        )
        .where(
            Diary.user_id == user_id,
            Diary.diary_date == target_date,
            Diary.deleted_at.is_(None)
        )
        .order_by(Diary.time_type)
    )
    result = await db.execute(stmt)
    return list(result.scalars().all())

def get_analysis_status(photo) -> str:
    """사진의 분석 상태 반환"""
    if photo.analysis_result is None:
        return "pending"
    return "done"

def get_diary_analysis_status(diary) -> str:
    """다이어리의 분석 상태 반환"""
    if not diary.photos:
        return "pending"

    all_done = all(
        p.analysis_result is not None
        for p in diary.photos
    )

    if all_done:
        return "done"
    elif any(p.analysis_result is not None for p in diary.photos):
        return "processing"
    else:
        return "pending"
```

### 2.3 API 라우터

**파일:** `app/routers/diaries.py`

```python
from fastapi import APIRouter, Depends
from sqlalchemy.ext.asyncio import AsyncSession
from datetime import date

from app.core.database import get_session
from app.core.dependencies import get_current_user
from app.models import User
from app.schemas.diary import DiariesByDateResponse, DiaryWithPhotos, PhotoInDiary
from app.services.diary_service import (
    get_diaries_by_date,
    get_analysis_status,
    get_diary_analysis_status
)

router = APIRouter(prefix="/diaries", tags=["diaries"])

@router.get("/{target_date}", response_model=DiariesByDateResponse)
async def get_diaries_by_date_endpoint(
    target_date: date,
    db: AsyncSession = Depends(get_session),
    current_user: User = Depends(get_current_user)
):
    """특정 날짜의 모든 다이어리 조회"""
    diaries = await get_diaries_by_date(db, current_user.id, target_date)

    diary_responses = []
    for diary in diaries:
        photos = [
            PhotoInDiary(
                photo_id=p.id,
                image_url=p.image_url,
                analysis_status=get_analysis_status(p)
            )
            for p in diary.photos
        ]

        diary_responses.append(DiaryWithPhotos(
            diary_id=diary.id,
            time_type=diary.time_type,
            restaurant_name=diary.restaurant_name,
            category=diary.category,
            analysis_status=get_diary_analysis_status(diary),
            cover_photo_url=diary.cover_photo.image_url if diary.cover_photo else None,
            photos=photos
        ))

    return DiariesByDateResponse(
        date=target_date,
        diaries=diary_responses
    )
```

from app.models import DiaryAnalysis

async def get_diary_by_id(
    db: AsyncSession,
    diary_id: int,
    user_id: UUID
) -> Diary | None:
    """다이어리 ID로 조회 (소유자 확인 포함)"""
    stmt = select(Diary).where(
        Diary.id == diary_id,
        Diary.user_id == user_id,
        Diary.deleted_at.is_(None)
    )
    result = await db.execute(stmt)
    return result.scalar_one_or_none()

async def get_diary_analysis(
    db: AsyncSession,
    diary_id: int,
    user_id: UUID
) -> DiaryAnalysis | None:
    """다이어리의 AI 분석 후보 조회"""
    # 먼저 다이어리 소유권 확인
    diary = await get_diary_by_id(db, diary_id, user_id)
    if diary is None:
        return None

    stmt = select(DiaryAnalysis).where(
        DiaryAnalysis.diary_id == diary_id
    )
    result = await db.execute(stmt)
    return result.scalar_one_or_none()

## 🔄 API 4: POST /diaries/{diary_id}/confirm - 다이어리 확정

```python
from app.schemas.diary import DiaryConfirm

async def confirm_diary(
    db: AsyncSession,
    diary_id: int,
    user_id: UUID,
    confirm_data: DiaryConfirm
) -> Diary:
    """유저가 AI 추정 결과를 확정합니다."""
    diary = await get_diary_by_id(db, diary_id, user_id)

    if diary is None:
        raise ValueError("Diary not found or access denied")

    # 확정 데이터 적용
    if confirm_data.restaurant_name is not None:
        diary.restaurant_name = confirm_data.restaurant_name
    if confirm_data.category is not None:
        diary.category = confirm_data.category

    await db.commit()
    await db.refresh(diary)

    return diary
```
